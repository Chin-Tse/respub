[log-format]
  srcip=3
  sport=4
  dstip=5
  dport=6
  proto_name=7
  proto_num=8
  ifname=9
  rxpkt=11
  rxsize=12
  txpkt=14

  agrp1=dstip, dport | rxpkt | 1000
